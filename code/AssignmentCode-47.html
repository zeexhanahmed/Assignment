Q 56: To search a given number in an integer array of 10 element (Linear Search)

                        
<script>
var userInput = prompt("Please Input Any no.");
userInput = parseInt(userInput);
var integerArray = [11, 2, 13, 4, 15, 6, 17, 8, 19, 10];
var numFound = false;

for (var i = 0; i <= integerArray.length - 1; i++) {

    if (userInput === integerArray[i]) {
        numFound = true;
        alert("Number Found!");
    }
}

if (numFound === false) {
    alert("Number Not Found!");
}
</script>
