Q 36: To calculate the sum of all number from 1 to N, at the end user will be asked whether to continue or not

                        
<script>
var total = 0;
var seriesSum = [];

do {
    var userInput = prompt("Please Enter any number");
    for(var i = 1; i <= userInput; i++) {
        total = total + i;
    }

    seriesSum.push(total);
    total = 0; 
    userInput2 =  prompt("If you want to continue press Y / N");
    userInput2 = userInput2.toUpperCase();

}while(userInput2 === "Y"); 
   
for(var j = 0; j <= seriesSum.length-1; j++) {
    document.write("The Sum of series = " + seriesSum[j] + "<br/>");
}
</script>
