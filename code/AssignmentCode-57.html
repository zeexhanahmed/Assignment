Q 66: To reverse each word of an inputted multi word sring

                        
<script>
var userInput = prompt("Please Write Any Sentence");
document.write("Before : " + userInput + "<br/>");
var inputArray = [];
inputArray = userInput.split(" ");

document.write("After : "); 

for(var i = 0; i < inputArray.length; i++) {

    for(var j = inputArray[i].length-1; j>=0; j--) {
        document.write(inputArray[i].charAt(j));
    }
    document.write(" ");
}
</script>
