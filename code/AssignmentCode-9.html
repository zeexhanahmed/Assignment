Q 11: To convert an inputted alphabet from upper to lower case and from lower to upper case

                        
<script>
var uppercaseInput = prompt("Please Input Any Word In Uppercase To Convert Into Lower");
var upperToLower = uppercaseInput.toLowerCase();
document.write("Here it's in Lowercase :" + " " + upperToLower + "<br/>");

var lowercaseInput = prompt("Please Input Any Word In Lowercase To Convert Into Upper");
var lowerToUpper = lowercaseInput.toUpperCase();
document.write("Here it's in Uppercase :" + " " + lowerToUpper);
</script>
