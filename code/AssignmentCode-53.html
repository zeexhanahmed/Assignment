Q 62: To check whether input string is palindrome or not

                        
<script>
var userInput = prompt("Please input any word to check whether its palindrome or not");
var flag = false;

for (var i = 0; i < userInput.length; i++) {
    if (userInput.charAt(i) !== userInput.charAt(userInput.length-i-1)) {
        flag = true;
    }
}

if (!flag) {
    document.write("It is Palindrome.");
} else {
    document.write("It is not Palindrome.");
}
</script>
