Q 63: To input a string and display it without displaying vowels

                        
<script>
var userInput = prompt("Please input any word");
userInput = userInput.split(" ");

var vowels = ["a", "e", "i", "o", "u"];
var userInputLower;

for (var i = 0; i < userInput.length; i++) {

    for (k = 0; k < userInput[i].length; k++) {
        
        userInputLower = userInput[i][k].toLowerCase();
        var flag = false;

        for (var j = 0; j < vowels.length; j++) {

            if (userInputLower == vowels[j]) {
                flag = true;
                break;
            }

        }

        if (!flag) {
            document.write(userInput[i][k]);
        }

    }
}
</script>
