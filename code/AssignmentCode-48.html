Q 57: To sort given integer array of 10 element in ascending order

                        
<script>
var integerArray = [5, 6, 7, 8, 9, 10, 15, 16, 13, 12, 11];
var intJ;
var intI;

document.write("Before: " + integerArray);
document.write("<br/>");

for(var i = 0; i < integerArray.length; i++) {
    for (var j= i+1; j < integerArray.length    ; j++) {
        if(integerArray[i] > integerArray[j]){ 
            
            intJ = integerArray[j];
            intI = integerArray[i];
            integerArray[i] = intJ;
            integerArray[j] = intI;

        }
    }
}
document.write("After: " + integerArray);
</script>
