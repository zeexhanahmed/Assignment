Q 30: To calculate the sum of input integer until user input negative no.

                        
<script>
var total = 0;
        
do {
    var userInput = prompt("Enter only Possitive number");
        userInput = parseInt(userInput);
    

    if(userInput >= 0) {
        total = total + userInput
    }

}while(userInput >= 0);

alert(" Oops! You have entered Negative number, Here is your Sum = " + total );
</script>
