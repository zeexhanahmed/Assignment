Q9: To check whether the input character is uppercase or lowercase

                        
<script>
var userinput = prompt("Please Input Any Word");


var updatedStringLower = userinput.toLowerCase();
var updatedStringUpper = userinput.toUpperCase();

if(userinput === updatedStringLower) {
    document.write("Your input is in lower case.");
} else if(userinput === updatedStringUpper) {
    document.write("Your input is in upper case.");
} else {
    document.write("Your input is mixed case.");
}
</script>
