Q8: To check whether an input year is leap year or not

                        
<script>
var userinput = prompt("Input Any Year");
        
function leapyear(userinput) {
    if (userinput % 4 === 0 && userinput % 100 !==0){
        return true;
    }else if(userinput % 400 === 0){
        return true;
    }else{
        return false;
    }

}

var answer = leapyear(userinput);
if (answer === true) {
    alert("It's a leap year");
}else{
    alert("It's not a leap year");
}
</script>
