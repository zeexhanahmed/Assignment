Q 10: To check whether the input alphabet is vowel or not

                        
<script>
var userinput = prompt("Please Enter Any Digit To Check Vowel");
var vowel = ["a", "e", "i", "o", "u"];

var matchFound = false;
for (var i = 0; i <= 4; i++){
    if(userinput === vowel[i]) {
        matchFound = true;
        alert("Its a vowel");   
    }
}

if (matchFound === false) {
    alert("Its not a vowel");
}
</script>
