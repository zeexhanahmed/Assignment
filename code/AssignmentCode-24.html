Q 35: Take a five digit number and print the reverse on the next line

                        
<script>
var userInput = prompt("Enter Only 5 digit or less");
        
if(userInput.length <= 5) {
    for (i = 1; i <= userInput.length; i++) {
    document.write(userInput.charAt(userInput.length-i) + "<br/>");
    }
}
else {
    alert("You inputted more than 5 digits");
}
</script>
